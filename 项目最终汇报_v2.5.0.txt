🎯 三语言使用示例
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
script {
    // ======================================
    // JavaScript 部分 - 数据和逻辑
    // ======================================
    
    const state = { count: 0 };
    
    function increment() {
        state.count++;
        return state.count;
    }
    
    // ======================================
    // CHTL JS 部分 - DOM 选择
    // ======================================
    
    const counter = {{#counter}};
    const button = {{.inc-btn}};
    
    // ======================================
    // 混合部分 - 事件处理
    // ======================================
    
    button &-> click: () => {
        const newCount = increment();  // JS 函数
        counter->textContent = newCount;  // CHTL JS
        console.log('Count:', newCount);  // JS
    };
    
    // ======================================
    // JavaScript 部分 - 异步操作
    // ======================================
    
    async function loadData() {
        const res = await fetch('/api/data');
        const data = await res.json();
        {{#output}}->innerHTML = JSON.stringify(data);
    }
}
